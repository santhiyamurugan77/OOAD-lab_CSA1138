#include <iostream>
#include <string>
using namespace std;

// =======================
// Admin Class
// =======================
class Admin {
public:
    int studentDetails;
    int bookList;

    void verifyStudentID() {
        cout << "Admin is verifying student ID..." << endl;
    }

    void checkAvailability() {
        cout << "Checking book availability..." << endl;
    }

    void issueBooks() {
        cout << "Book issued to student." << endl;
    }

    void orderNewAuthor() {
        cout << "Order placed for books of a new author." << endl;
    }

    void maintainStudentDetails() {
        cout << "Maintaining student details..." << endl;
    }
};

// =======================
// Student Class
// =======================
class Student {
public:
    int studentDetails;

    void requestBooks() {
        cout << "Student requested a book." << endl;
    }

    void registerStudent() {
        cout << "Student registered successfully." << endl;
    }
};

// =======================
// Computer Class
// =======================
class Computer {
public:
    int studentRecord;
    int bookList;

    void maintainStudentRecord() {
        cout << "Computer maintaining student records..." << endl;
    }

    void enterIssue() {
        cout << "Computer recorded the book issue entry." << endl;
    }

    void orderNewAuthor() {
        cout << "Computer ordered new author books." << endl;
    }

    void checkAvailability() {
        cout << "Computer checking book availability..." << endl;
    }
};

// =======================
// Main Function
// =======================
int main() {
    // Admin object
    Admin admin;
    admin.verifyStudentID();
    admin.checkAvailability();
    admin.issueBooks();
    admin.orderNewAuthor();
    admin.maintainStudentDetails();

    cout << endl;

    // Student object
    Student s;
    s.registerStudent();
    s.requestBooks();

    cout << endl;

    // Computer object
    Computer c;
    c.maintainStudentRecord();
    c.enterIssue();
    c.orderNewAuthor();
    c.checkAvailability();

    return 0;
}
